//捕获来源192.168.0.1的数据
src host 192.168.0.1 
//捕获目的192.168.0.1的数据
dst host 192.168.0.1 
//捕获来源或者目的为192.168.0.1的数据
host 192.168.0.1

//捕获来源192.168.0.1的ip数据
ip src host 192.168.0.1 
//捕获目的192.168.0.1的ip数据
ip dst host 192.168.0.1 
//捕获来源或者目的为192.168.0.1的ip数据
ip host 192.168.0.1

//捕获来源mac为xx:xx:xx:xx:xx:xx的数据
ether src  xx:xx:xx:xx:xx:xx
//捕获目的mac为xx:xx:xx:xx:xx:xx的数据
ether dst  xx:xx:xx:xx:xx:xx
//捕获来源或目的mac为xx:xx:xx:xx:xx:xx的数据
ether host  xx:xx:xx:xx:xx:xx

//捕获目的端口为8000的数据
dst port  8000
//捕获源端口为8000的数据
src port  8000
//捕获源、目的端口为8000的数据
port  8000

//捕获源端口为8000的TCP数据
tcp src port 8000
//捕获目的端口为8000的TCP数据
tcp dst port 8000
//捕获ip或者arp
ip or arp
