	2)获得键值函数

	还可以自己封装一个无阻塞，无回显的getch，实现如下：
	#include <termios.h>
	#include <unistd.h>
	char mygetch()
	 {
		struct termios oldt, newt;
		char ch;
		tcgetattr( STDIN_FILENO, &oldt );
		newt = oldt;
		newt.c_lflag &= ~( ICANON | ECHO );
		tcsetattr( STDIN_FILENO, TCSANOW, &newt );
		ch = getchar();
		tcsetattr( STDIN_FILENO, TCSANOW, &oldt );
		return ch;
	}
